最新版Vue3实践教程

一、创建项目

# 创建项目
npm init vite-app <projectname>
# 或者
yarn create vite my-vue --template vue
# 安装依赖
yarn install
# 启动项目
yarn dev

yarn add element-plus

import ElementPlus from 'element-plus'

三、VueRouter

安装

cnpm install vue-router@4

在src目录下创建router文件夹并新建index.js文件

import {createRouter,createWebHashHistory} from 'vue-router'

const routes = [
  {
    path:'/',
    component: () => import('../views/Main.vue'),
    children:[
      {
        path:'home',
        name:'Home',
        component: () => import('../views/home/Home.vue')
      }
    ]
  },
]

const router = createRouter({
  history:createWebHashHistory(),
  routes
})

export default router

在main.js中引入

import router from './router' //这里用的是ts 出现报红需要npm install --save-dev @types/js-md5
app.use(router)

使用

<!--模板语法中使用-->
{{$router.push('/region')}}
{{$route.params.name}}

<!--js中使用-->
<script>
//引入useRouter
import {useRouter,useRoute} from 'vue-router'
export default {
    setup(){
        const router = useRouter()
        const route = useRoute()
        //编程式路由导航
        //使用参数，要在router/index.js配置路由时参数占位
        router.push({
            name:'Region',
            params:{
                name:'张三',
                age:23
            }
        })
    }
}
</script>

五、Vuex

安装

cnpm install vuex -S

在src目录下新建store文件并创建index.js

import {createStore} from 'vuex'

export default createStore({
  state:{
    isCollapse:true
  },
  actions:{
    changeIsCollapse(context,value){
        context.commit('CHANGEISCOLLAPSE',value)
    }
  },
  mutations:{
    CHANGEISCOLLAPSE(state,value){
      state.isCollapse = !state.isCollapse
    }
  },
  getters:{
      handleCollapse(state){
          //处理state中数据
      }
  }
})

在main.js中使用

import store from './store'

app.use(store)

使用

<!--模板语法使用-->
{{$store.state.isCollapse}}

<script>
//js中使用
//必须引入
import { useStore } from 'vuex';
    export default {
        setup() {
            //在这里使用useStore函数赋值给store
            let store = useStore()
            //改变侧边栏状态
            let handleCollapse = () => {
                store.commit('changeIsCollapse',value)
                store.commit('CHANGEISCOLLAPSE',value)
            }

            return {
                handleCollapse
            }
        }
    }
</script>

七、vite跨域配置

在根目录下创建vite.config.js

import {defineConfig} from 'vite'
import vue from '@vitejs/plugin-vue'

// https://vitejs.dev/config/
export default defineConfig({
    server: {
        proxy: {
            '/api': {
                target: 'http://localhost:8081',
                changeOrigin: true,  // 允许跨域
                rewrite: path => path.replace(/^\/api/, '')

            '/map': {

                changeOrigin: true,  // 允许跨域
                rewrite: path => path.replace(/^\/map/, '')
            }
        },
    },
    plugins: [
        vue(),
    ],
})

八、二次封装axios

在src目录下创建api文件夹并新建request.js

import axios from 'axios'
// import Router from "vue-router";
import { ElMessage } from 'element-plus'

const request = axios.create({
  // baseURL: '/api',  //设计到多个跨域路径，不写baseURL，那么在发送请求时就需要加上前缀了
  timeout: 10000
})

// request 拦截器
// 可以自请求发送前对请求做一些处理
// 比如统一加token，对请求参数统一加密
let user = null
request.interceptors.request.use(config => {
  config.headers['Content-Type'] = 'application/json;charset=utf-8';
  user = localStorage.getItem("user") ? JSON.parse(localStorage.getItem("user")) : null;
  if (user) {
    config.headers['token'] = user.token;  // 设置请求头
  }
  return config
}, error => {
  return Promise.reject(error)
});

// response 拦截器
// 可以在接口响应后统一处理结果
request.interceptors.response.use(
  response => {
    let res = response.data;
    // 如果是返回的文件
    if (response.config.responseType === 'blob') {
      return res
    }
    // 兼容服务端返回的字符串数据
    if (typeof res === 'string') {
      res = res ? JSON.parse(res) : res
    }
    if (res.code === 0) {
      console.log(res)
      ElMessage({
        message: msg,
        type: 'error',
      })
      // Router.push("/login")
    }
    return res;
  },
  error => {
    console.log('err' + error) // for debug
    return Promise.reject(error)
  }
)

export default request

在main.js文件中挂载到全局

import request from './api/request'

//挂载方式和vue2不同 使用app.config.globalProperties
app.config.globalProperties.request = request

使用

    params:{
        name:'乐山师范'
    }
}).then(res => {
    if(res.code === 1){
        console.log(res.data);
    }
})
})

十一、Pinia

安装

npm install pinia

在main.js中引入

import { createPinia } from 'pinia'
const pinia = createPinia()
const app = createApp(App)
app.use(pinia)

在src目录下创建store并新建index.js

import { defineStore} from 'pinia'

export const useCounterStore  = defineStore('counter',{
  state:()=>{
    return {
      count:0,
    }
  },
  getters:{
      double:(state) => state.count * 2,
  },
  actions:{
    increment(){
      this.count ++
    }
  }
})

export const useUserStore  = defineStore('user',{
  state:()=>{
    return {	
      isMale:true
    }
  },
  getters:{},
  actions:{
    changeSex(){
      this.isMale = !this.isMale
    }
  }
})

使用

<script>
import {useCounterStore,useUserStore } from '../store'
import {storeToRefs } from 'pinia'
export default {
  name:'Home',
  setup(){
    const countStore = useCounterStore()
    const userStore = useUserStore()
    //通过解构赋值 得到count并且将其设置为响应式的
    // const {count} = storeToRefs(countStore)

    const handleCount = () => {
        //直接修改state中的count
      // countStore.count++
      //调用actions中的increment函数
      countStore.increment函数()
      //通过$patch()里面传递函数来修改state中的count
      countStore.$patch(state => {
          state.count ++
      })
    }
    const handleSex = () => {
      //调用actions中的changeSex函数
      userStore.changeSex()
    }


    return {
      handleCount,countStore,userStore,handleSex
    }
  }
}
</script>

十二、setup语法糖–基本

1.父传子传参–defineProps

声明参数类型

<!--父组件-->
<template>
	<Child msg="子元素，你在干嘛"></Child>
</template>

<!--子组件-->
<template>
	我是子元素，这是父元素给我带的话{{props.msg}}
</template>
<script setup>
import {defineProps} from 'vue'
// 在接收时候也得注意，vue3 props接收必须规定数据类型，如果父元素数据类型出错，那么会报错
const props = defineProps({msg:String})
</script>

定义接口，规范格式

<!--父组件-->
<template>
<Child :msg="msg" :count="count" :user-list="userList"/>
</template>
<script setup lang="ts">
import {ref} from "vue";
import Child from "./components/Child.vue";

const msg = ref<String>('帅')
const count = ref<Number>(9527)
const userList = ref([
  {name: '张三' as string},
  {name: '李四' as string},
  {name: '王五' as string},
  {name: '赵六' as string},
])
</script>

<!------上面父，下面子-------->

<!--子组件-->
<template>
{{ props.msg}}
{{ props.count}}
  <div v-for="(item,index) in props.userList" :key="index">
    {{item.name}}
  </div>
</template>

<script setup lang="ts">
import {defineProps} from "vue";

interface Props {
  msg: String,
  count: Number,
  userList: Array<String>
}

const props = defineProps<Props>()

</script>

withDefaults

类似于Java中的Optional可选类中的ofNullable方法，可以设置默认值

<script setup lang="ts">
import {defineProps,withDefaults} from "vue";

interface Props {
  msg: String,
  count: Number,
  userList: Array<String>
}

// const props = defineProps<Props>()
const props = withDefaults(defineProps<Props>(),{
  msg: () => '',
  count: () => 0,
  userList: () => []
})

import mitt from 'mitt'
const bus = mitt()
export default bus

3.使用

  • 提供数据方
import bus from "../util/bus.js";

bus.emit('getHeritageList',heritageStore.searchInputValue)
  • 接收数据方
import bus from "../util/bus.js";

//挂载
onMounted(() => {
  changeConditionList()
  bus.on('getHeritageList',() => {
    initHeritageList()

})
//销毁之前
onBeforeUnmount( () => {
